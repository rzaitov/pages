def to_timedelta(x):
    arr = x.split(':')
    return timedelta(minutes=int(arr[0]), seconds=float(arr[1]))

def to_xy(data_frame, dist_column, time_column):
    rows = data_frame[time_column].notnull()
    return (data_frame[dist_column][rows], data_frame[time_column][rows].map(to_timedelta) / pd.Timedelta(seconds=1))

data1 = pd.read_csv('swim_2017_03_05.csv')
dist_1_1, seconds_1_1 = to_xy(data1, 'dist', 'time1')
dist_1_2, seconds_1_2 = to_xy(data1, 'dist', 'time2')
dist_1_3, seconds_1_3 = to_xy(data1, 'dist', 'time3')

plt.plot(dist_1_1, seconds_1_1, label='первый')
plt.plot(dist_1_2, seconds_1_2, label='второй')
plt.plot(dist_1_3, seconds_1_3, label='третий')
plt.xlabel('дистанция')
plt.ylabel('время [сек]')
plt.title('5 марта 2017')
plt.legend()
